CREATE DATABASE projectSS10;


CREATE TABLE users(
id INT AUTO_INCREMENT PRIMARY KEY,
username VARCHAR(18) NOT NULL,
pwd CHAR(32) NOT NULL,
level TINYINT UNSIGNED NOT NULL DEFAULT 0,
addtime INT UNSIGNED NOT NULL)ENGINE=MYISAM DEFAULT CHARSET=UTF8;



 CREATE TABLE user_info(
 uid INT PRIMARY KEY,
 name VARCHAR(30) NOT NULL,
 sex TINYINT UNSIGNED NOT NULL DEFAULT 0,
 age TINYINT UNSIGNED NOT NULL,
 phone CHAR(11) NOT NULL,
 email VARCHAR(50) NOT NULL,
 xueli TINYINT UNSIGNED NOT NULL,
 shouru TINYINT UNSIGNED NOT NULL,
 pic CHAR(37) NOT NULL,
 address VARCHAR(255) NOT NULL,
 hunfou TINYINT UNSIGNED NOT NULL)ENGINE=MYISAM DEFAULT CHARSET=UTF8;
